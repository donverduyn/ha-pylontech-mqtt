# Single source of truth for every AI CLI config path that needs to survive
# a container rebuild, read by four scripts (host-side seedHostAgentConfig.sh
# and verifyHostAgentConfig.sh, container-side postCreate.sh and
# syncAgentConfigOut.sh) so the list is never duplicated. Format: "<path
# relative to $HOME>|<kind>", where kind is:
#   dir   - a whole directory, bind-mounted read-write directly from this
#           project's backup onto the container's real path for it (see
#           devcontainer.json's "mounts"). Seeded once, ever, from whatever
#           the host has at that path; after that the mount itself is the
#           only copy anyone reads or writes -- no copying, no polling.
#   json  - a bare file, not inside its own directory, so it can't be
#           bind-mounted the same way (see seedHostAgentConfig.sh for why).
#           Still handled the old file-by-file way: seeded once as {} if
#           the host doesn't have one, copied into the container by
#           postCreate.sh, and polled back out by syncAgentConfigOut.sh.

# Claude Code
.claude|dir
.claude.json|json

# Codex
.codex|dir

# OpenCode
.config/opencode|dir
.local/share/opencode|dir

# Kilocode CLI
.config/kilo|dir

# GitHub CLI
.config/gh|dir

# Antigravity CLI / Gemini CLI (shared ~/.gemini tree)
.gemini|dir

# GitHub Copilot CLI (session-store.db mixes login in with everything else
# it stores, same reasoning as the other directories above)
.copilot|dir
